

- 打开自动创建索引

```sh
PUT /_cluster/settings
{
  "persistent": {
    "action.auto_create_index": "true"
  }
}
```



- 打开自动创建索引白名单

```sh
PUT /_cluster/settings
{
  "persistent": {
    "action.auto_create_index": "cloud_*"
  }
}
```



- 创建索引模板

```sh
POST /_index_template/cloud_demo_work_record_template
{
    "index_patterns": [
        "cloud_demo_work_record_doc_*"
    ],
    "template": {
        "settings": {
            "index.number_of_shards": 3,
            "index.number_of_replicas": 3
        },
		"aliases": {
          "cloud_demo_work_record_alias":{}
        },
        "mappings": {
            "dynamic": "true",
            "properties": {
                "id": {
                    "type": "keyword"
                },
                "deviceSn": {
                    "type": "keyword"
                },
                "type": {
                    "type": "keyword"
                },
                "deleted": {
                    "type": "keyword"
                },
                "eventType": {
                    "type": "keyword"
                },
                "content": {
                    "type": "keyword"
                },
                "params": {
                    "type": "keyword"
                },
                "alarmTag": {
                    "type": "boolean"
                },
                "recordTime": {
                    "type": "long"
                },
                "createTime": {
                    "type": "long"
                }
            }
        }
    }
}
```



- 旧索引reindex至新索引

```sh
POST /_reindex
{
  "source": {
    "index": "cloud_demo_work_record_doc",
    "query": {
      "bool": {
        "filter": {
          "range": {
            "createTime": {
              "gte": "1688140800000",
              "lt": "1690819200000"
            }
          }
        }
      }
    }
  },
  "dest": {
    "index": "cloud_demo_work_record_doc_2023_07"
  }
}
```
